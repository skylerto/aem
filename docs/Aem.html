def read file="#{ENV["HOME"]}/.aem.yaml"
  if File.exist?(file)
    return YAML.load_file(file)
  elsif File.exist?("#{ENV["HOME"]}/#{file}")
    return YAML.load_file("#{ENV["HOME"]}/#{file}")
  else
    return nil
  end
end
